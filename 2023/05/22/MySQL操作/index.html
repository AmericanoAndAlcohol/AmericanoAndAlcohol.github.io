查找

WHERE 子句的字符串比较是不区分大小写的。 你可以使用 BINARY关键字来区分大小写
SELECT * from runoob_tbl WHERE runoob_author=’菜鸟教程’;
SELECT * from runoob_tbl WHERE BINARY runoob_author=’runoob.com’;
